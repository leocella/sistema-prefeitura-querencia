# Sistema de Gestão de Exames Laboratoriais — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a complete web system for managing lab exam queues for Prefeitura de Querência/MT, deployed on Vercel with Supabase backend.

**Architecture:** Next.js 14 App Router with Server Components and Server Actions for data fetching/mutations. Supabase handles PostgreSQL (with RLS), Auth (email/password), Storage (PDF laudos), and Realtime (dashboard live updates). Authentication is enforced via Next.js middleware.

**Tech Stack:** Next.js 14, TypeScript, Tailwind CSS, shadcn/ui, Supabase (SSR), Vitest + React Testing Library, Sonner (toasts), Vercel

  ```sql
  -- supabase/migrations/001_schema.sql

  -- Prestadores de serviço (4 fixos)
  create table if not exists prestadores (
    id uuid primary key default gen_random_uuid(),
    nome text not null,
    ativo boolean default true,
    created_at timestamptz default now()
  );

  -- Pacientes
  create table if not exists pacientes (
    id uuid primary key default gen_random_uuid(),
    nome_completo text not null,
    data_nascimento date not null,
    cpf text unique,
    created_at timestamptz default now()
  );

  -- Solicitações (entidade principal)
  create table if not exists solicitacoes (
    id uuid primary key default gen_random_uuid(),
    codigo_barras text unique not null,
    paciente_id uuid references pacientes(id),
    prestador_id uuid references prestadores(id),
    medico_solicitante text not null,
    solicitante_origem text,
    data_prevista date not null,
    hora_prevista time,
    classificacao text check (classificacao in ('internamento', 'emergencia')) not null,
    status text check (status in ('solicitado', 'coletado', 'liberado')) default 'solicitado',
    qtd_exames integer default 0,
    qtd_realizados integer default 0,
    laudo_url text,
    laudo_uploaded_at timestamptz,
    created_at timestamptz default now(),
    updated_at timestamptz default now()
  );

  -- Exames individuais
  create table if not exists exames (
    id uuid primary key default gen_random_uuid(),
    solicitacao_id uuid references solicitacoes(id) on delete cascade,
    nome_exame text not null,
    codigo_procedimento text,
    realizado boolean default false,
    created_at timestamptz default now()
  );

  -- Usuários (mirrors auth.users)
  create table if not exists usuarios (
    id uuid references auth.users(id) primary key,
    nome text not null,
    prestador_id uuid references prestadores(id),
    role text check (role in ('admin', 'laboratorio', 'hospital')) default 'laboratorio'
  );

  -- Auto-update updated_at
  create or replace function update_updated_at()
  returns trigger as $$
  begin
    new.updated_at = now();
    return new;
  end;
  $$ language plpgsql;

  create trigger solicitacoes_updated_at
    before update on solicitacoes
    for each row execute function update_updated_at();

  -- Row Level Security
  alter table prestadores enable row level security;
  alter table pacientes enable row level security;
  alter table solicitacoes enable row level security;
  alter table exames enable row level security;
  alter table usuarios enable row level security;

  -- prestadores: authenticated users can read
  create policy "prestadores_read" on prestadores
    for select to authenticated using (true);

  -- pacientes: authenticated users can read and insert
  create policy "pacientes_read" on pacientes
    for select to authenticated using (true);

  create policy "pacientes_insert" on pacientes
    for insert to authenticated with check (true);

  -- usuarios: users can read their own record
  create policy "usuarios_read_own" on usuarios
    for select to authenticated using (auth.uid() = id);

  -- solicitacoes: laboratorio sees only their prestador; admin/hospital see all
  create policy "solicitacoes_select" on solicitacoes
    for select to authenticated using (
      (select role from usuarios where id = auth.uid()) in ('admin', 'hospital')
      or prestador_id = (select prestador_id from usuarios where id = auth.uid())
    );

  create policy "solicitacoes_update" on solicitacoes
    for update to authenticated using (
      (select role from usuarios where id = auth.uid()) in ('admin', 'hospital')
      or prestador_id = (select prestador_id from usuarios where id = auth.uid())
    );

  create policy "solicitacoes_insert" on solicitacoes
    for insert to authenticated with check (true);

  -- exames: same access rules as parent solicitacao
  create policy "exames_select" on exames
    for select to authenticated using (
      exists (
        select 1 from solicitacoes s
        where s.id = solicitacao_id
        and (
          (select role from usuarios where id = auth.uid()) in ('admin', 'hospital')
          or s.prestador_id = (select prestador_id from usuarios where id = auth.uid())
        )
      )
    );

  create policy "exames_update" on exames
    for update to authenticated using (
      exists (
        select 1 from solicitacoes s
        where s.id = solicitacao_id
        and (
          (select role from usuarios where id = auth.uid()) in ('admin', 'hospital')
          or s.prestador_id = (select prestador_id from usuarios where id = auth.uid())
        )
      )
    );

  create policy "exames_insert" on exames
    for insert to authenticated with check (true);
  ```

## Task 7: Middleware (Route Protection)

**Files:**
- Create: `middleware.ts`

- [ ] **Step 1: Create middleware.ts**

  ```typescript
  // middleware.ts
  import { createServerClient } from '@supabase/ssr'
  import { NextResponse, type NextRequest } from 'next/server'

  export async function middleware(request: NextRequest) {
    let supabaseResponse = NextResponse.next({ request })

    const supabase = createServerClient(
      process.env.NEXT_PUBLIC_SUPABASE_URL!,
      process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
      {
        cookies: {
          getAll() {
            return request.cookies.getAll()
          },
          setAll(cookiesToSet) {
            cookiesToSet.forEach(({ name, value }) =>
              request.cookies.set(name, value)
            )
            supabaseResponse = NextResponse.next({ request })
            cookiesToSet.forEach(({ name, value, options }) =>
              supabaseResponse.cookies.set(name, value, options)
            )
          },
        },
      }
    )

    const {
      data: { user },
    } = await supabase.auth.getUser()

    const isProtectedRoute =
      request.nextUrl.pathname.startsWith('/dashboard') ||
      request.nextUrl.pathname.startsWith('/solicitacao')

    if (!user && isProtectedRoute) {
      return NextResponse.redirect(new URL('/login', request.url))
    }

    if (user && request.nextUrl.pathname === '/login') {
      return NextResponse.redirect(new URL('/dashboard', request.url))
    }

    return supabaseResponse
  }

  export const config = {
    matcher: ['/((?!_next/static|_next/image|favicon.ico|api).*)'],
  }
  ```

  ```tsx
  // components/StatusButtons.tsx
  'use client'

  import { useState } from 'react'
  import { Button } from '@/components/ui/button'
  import { cn } from '@/lib/utils'
  import { SolicitacaoStatus } from '@/lib/types'
  import { createClient } from '@/lib/supabase'
  import { toast } from 'sonner'

  interface StatusButtonsProps {
    solicitacaoId: string
    currentStatus: SolicitacaoStatus
    onStatusChange?: (newStatus: SolicitacaoStatus) => void
    onLiberadoClick?: () => void
    size?: 'sm' | 'default'
  }

  const STATUS_ORDER: SolicitacaoStatus[] = ['solicitado', 'coletado', 'liberado']

  const STATUS_LABEL: Record<SolicitacaoStatus, string> = {
    solicitado: 'Solicitado',
    coletado: 'Coletado',
    liberado: 'Liberado',
  }

  const STATUS_ACTIVE_CLASS: Record<SolicitacaoStatus, string> = {
    solicitado: 'bg-gray-200 text-gray-800 border-gray-400 hover:bg-gray-200',
    coletado: 'bg-amber-100 text-amber-800 border-amber-400 hover:bg-amber-100',
    liberado: 'bg-green-100 text-green-800 border-green-400 hover:bg-green-100',
  }

  export function StatusButtons({
    solicitacaoId,
    currentStatus,
    onStatusChange,
    onLiberadoClick,
    size = 'sm',
  }: StatusButtonsProps) {
    const [status, setStatus] = useState(currentStatus)
    const [loading, setLoading] = useState(false)
    const supabase = createClient()

    async function handleClick(newStatus: SolicitacaoStatus) {
      if (newStatus === status || loading) return

      const currentIdx = STATUS_ORDER.indexOf(status)
      const newIdx = STATUS_ORDER.indexOf(newStatus)
      if (newIdx <= currentIdx) return

      if (newStatus === 'liberado' && onLiberadoClick) {
        onLiberadoClick()
        return
      }

      setStatus(newStatus)
      onStatusChange?.(newStatus)
      setLoading(true)

      const { error } = await supabase
        .from('solicitacoes')
        .update({ status: newStatus })
        .eq('id', solicitacaoId)

      setLoading(false)

      if (error) {
        setStatus(status)
        toast.error('Erro ao atualizar status')
      } else {
        toast.success(`Status atualizado: ${STATUS_LABEL[newStatus]}`)
      }
    }

    return (
      <div className="flex gap-2" role="group" aria-label="Status da solicitação">
        {STATUS_ORDER.map((s) => {
          const isActive = status === s
          const isPast = STATUS_ORDER.indexOf(s) < STATUS_ORDER.indexOf(status)
          return (
            <Button
              key={s}
              variant={isActive ? 'default' : 'outline'}
              size={size}
              onClick={() => handleClick(s)}
              disabled={loading || isPast}
              className={cn(isActive && STATUS_ACTIVE_CLASS[s])}
              aria-pressed={isActive}
            >
              {STATUS_LABEL[s]}
            </Button>
          )
        })}
      </div>
    )
  }
  ```

  ```tsx
  'use client'

  import { useState } from 'react'
  import {
    Dialog,
    DialogContent,
    DialogHeader,
    DialogTitle,
    DialogFooter,
  } from '@/components/ui/dialog'
  import { Button } from '@/components/ui/button'
  import { Input } from '@/components/ui/input'
  import { Label } from '@/components/ui/label'
  import { createClient } from '@/lib/supabase'
  import { toast } from 'sonner'

  interface LaudoUploadProps {
    solicitacaoId: string
    laudoUrl: string | null
    open: boolean
    onOpenChange: (open: boolean) => void
    onUploaded: (url: string) => void
  }

  export function LaudoUpload({
    solicitacaoId,
    laudoUrl,
    open,
    onOpenChange,
    onUploaded,
  }: LaudoUploadProps) {
    const [file, setFile] = useState<File | null>(null)
    const [loading, setLoading] = useState(false)
    const supabase = createClient()

    async function handleSubmit() {
      if (!file) return
      setLoading(true)

      const path = `laudos/${solicitacaoId}/laudo.pdf`

      const { error: uploadError } = await supabase.storage
        .from('laudos')
        .upload(path, file, { upsert: true })

      if (uploadError) {
        toast.error('Erro ao enviar laudo')
        setLoading(false)
        return
      }

      const {
        data: { publicUrl },
      } = supabase.storage.from('laudos').getPublicUrl(path)

      const { error: updateError } = await supabase
        .from('solicitacoes')
        .update({
          laudo_url: publicUrl,
          laudo_uploaded_at: new Date().toISOString(),
          status: 'liberado',
        })
        .eq('id', solicitacaoId)

      setLoading(false)

      if (updateError) {
        toast.error('Erro ao salvar laudo')
        return
      }

      toast.success('Laudo enviado com sucesso')
      onUploaded(publicUrl)
      onOpenChange(false)
      setFile(null)
    }

    return (
      <Dialog open={open} onOpenChange={onOpenChange}>
        <DialogContent>
          <DialogHeader>
            <DialogTitle>Enviar Laudo</DialogTitle>
          </DialogHeader>

          <div className="space-y-4">
            {laudoUrl && (
              <Button variant="outline" asChild>
                <a href={laudoUrl} target="_blank" rel="noopener noreferrer">
                  Visualizar laudo atual
                </a>
              </Button>
            )}

            <div className="space-y-2">
              <Label htmlFor="laudo-input">Arquivo PDF</Label>
              <Input
                id="laudo-input"
                type="file"
                accept="application/pdf"
                onChange={(e) => setFile(e.target.files?.[0] ?? null)}
              />
            </div>
          </div>

          <DialogFooter>
            <Button variant="outline" onClick={() => onOpenChange(false)}>
              Cancelar
            </Button>
            <Button onClick={handleSubmit} disabled={!file || loading}>
              {loading ? 'Enviando...' : 'Enviar laudo'}
            </Button>
          </DialogFooter>
        </DialogContent>
      </Dialog>
    )
  }
  ```

  This is the client component that handles Realtime subscriptions and filter state.

  ```tsx
  // app/dashboard/DashboardClient.tsx
  'use client'

  import { useEffect, useState } from 'react'
  import { useRouter, useSearchParams } from 'next/navigation'
  import { createClient } from '@/lib/supabase'
  import { Solicitacao, Prestador, SolicitacaoStatus } from '@/lib/types'
  import { SolicitacaoCard } from '@/components/SolicitacaoCard'
  import {
    Select,
    SelectContent,
    SelectItem,
    SelectTrigger,
    SelectValue,
  } from '@/components/ui/select'
  import { Input } from '@/components/ui/input'

  interface DashboardClientProps {
    initialSolicitacoes: Solicitacao[]
    prestadores: Prestador[]
    userPrestadorId: string | null
  }

  export function DashboardClient({
    initialSolicitacoes,
    prestadores,
    userPrestadorId,
  }: DashboardClientProps) {
    const router = useRouter()
    const searchParams = useSearchParams()

    const today = new Date().toISOString().split('T')[0]
    const [solicitacoes, setSolicitacoes] = useState(initialSolicitacoes)
    const [data, setData] = useState(searchParams.get('data') ?? today)
    const [prestadorId, setPrestadorId] = useState(
      searchParams.get('prestador') ?? userPrestadorId ?? 'todos'
    )
    const [classificacao, setClassificacao] = useState(
      searchParams.get('classificacao') ?? 'todas'
    )

    const supabase = createClient()

    // Realtime subscription — refresh list on any change to solicitacoes
    useEffect(() => {
      const channel = supabase
        .channel('solicitacoes-changes')
        .on(
          'postgres_changes',
          { event: '*', schema: 'public', table: 'solicitacoes' },
          () => { router.refresh() }
        )
        .subscribe()

      return () => { supabase.removeChannel(channel) }
    }, [supabase, router])

    // Sync solicitacoes when server re-renders (router.refresh)
    useEffect(() => {
      setSolicitacoes(initialSolicitacoes)
    }, [initialSolicitacoes])

    const filtered = solicitacoes.filter((s) => {
      if (prestadorId !== 'todos' && s.prestador_id !== prestadorId) return false
      if (classificacao !== 'todas' && s.classificacao !== classificacao) return false
      return true
    })

    function handleDataChange(e: React.ChangeEvent<HTMLInputElement>) {
      setData(e.target.value)
      const params = new URLSearchParams(searchParams.toString())
      params.set('data', e.target.value)
      router.push(`/dashboard?${params.toString()}`)
    }

    return (
      <div className="space-y-4">
        {/* Filters */}
        <div className="flex flex-wrap gap-3">
          <Input
            type="date"
            value={data}
            onChange={handleDataChange}
            className="w-auto"
          />

          <Select value={prestadorId} onValueChange={setPrestadorId}>
            <SelectTrigger className="w-44">
              <SelectValue placeholder="Prestador" />
            </SelectTrigger>
            <SelectContent>
              <SelectItem value="todos">Todos</SelectItem>
              {prestadores.map((p) => (
                <SelectItem key={p.id} value={p.id}>{p.nome}</SelectItem>
              ))}
            </SelectContent>
          </Select>

          <Select value={classificacao} onValueChange={setClassificacao}>
            <SelectTrigger className="w-44">
              <SelectValue placeholder="Classificação" />
            </SelectTrigger>
            <SelectContent>
              <SelectItem value="todas">Todas</SelectItem>
              <SelectItem value="emergencia">Emergência</SelectItem>
              <SelectItem value="internamento">Internamento</SelectItem>
            </SelectContent>
          </Select>
        </div>

        {/* List */}
        {filtered.length === 0 ? (
          <p className="text-center text-muted-foreground py-12">
            Nenhuma solicitação encontrada para os filtros selecionados.
          </p>
        ) : (
          <div className="space-y-3">
            {filtered.map((s) => (
              <SolicitacaoCard key={s.id} solicitacao={s} />
            ))}
          </div>
        )}
      </div>
    )
  }
  ```

- [ ] **Step 2: Create app/dashboard/page.tsx**

  ```tsx
  // app/dashboard/page.tsx
  import { redirect } from 'next/navigation'
  import { createClient } from '@/lib/supabase-server'
  import { DashboardClient } from './DashboardClient'
  import { Button } from '@/components/ui/button'
  import { signOut } from '@/app/login/actions'

  export default async function DashboardPage({
    searchParams,
  }: {
    searchParams: { data?: string; prestador?: string; classificacao?: string }
  }) {
    const supabase = createClient()

    const { data: { user } } = await supabase.auth.getUser()
    if (!user) redirect('/login')

    // Get current user profile
    const { data: usuario } = await supabase
      .from('usuarios')
      .select('*')
      .eq('id', user.id)
      .single()

    // Get prestadores list
    const { data: prestadores } = await supabase
      .from('prestadores')
      .select('*')
      .eq('ativo', true)
      .order('nome')

    // Default date = today
    const today = new Date().toISOString().split('T')[0]
    const data = searchParams.data ?? today

    // Fetch solicitacoes for the selected date
    let query = supabase
      .from('solicitacoes')
      .select('*, pacientes(*), prestadores(*)')
      .eq('data_prevista', data)
      .order('hora_prevista', { ascending: true })

    const { data: solicitacoes } = await query

    return (
      <div className="min-h-screen bg-gray-50">
        {/* Header */}
        <header className="bg-white border-b px-4 py-3 flex items-center justify-between">
          <div>
            <h1 className="font-semibold text-lg">Lab Querência</h1>
            {usuario && (
              <p className="text-sm text-muted-foreground">{usuario.nome}</p>
            )}
          </div>
          <form action={signOut}>
            <Button variant="outline" size="sm" type="submit">
              Sair
            </Button>
          </form>
        </header>

        <main className="max-w-2xl mx-auto p-4">
          <DashboardClient
            initialSolicitacoes={solicitacoes ?? []}
            prestadores={prestadores ?? []}
            userPrestadorId={usuario?.prestador_id ?? null}
          />
        </main>
      </div>
    )
  }
  ```

  ```tsx
  // app/solicitacao/[id]/SolicitacaoClient.tsx
  'use client'

  import { useState } from 'react'
  import { useRouter } from 'next/navigation'
  import { Solicitacao } from '@/lib/types'
  import { StatusButtons } from '@/components/StatusButtons'
  import { LaudoUpload } from '@/components/LaudoUpload'
  import { createClient } from '@/lib/supabase'
  import { toast } from 'sonner'
  import { Checkbox } from '@/components/ui/checkbox'
  import { Label } from '@/components/ui/label'

  interface SolicitacaoClientProps {
    solicitacao: Solicitacao
  }

  export function SolicitacaoClient({ solicitacao }: SolicitacaoClientProps) {
    const router = useRouter()
    const supabase = createClient()
    const [laudoOpen, setLaudoOpen] = useState(false)
    const [laudoUrl, setLaudoUrl] = useState(solicitacao.laudo_url)

    async function handleExameToggle(exameId: string, realizado: boolean) {
      const { error } = await supabase
        .from('exames')
        .update({ realizado })
        .eq('id', exameId)

      if (error) {
        toast.error('Erro ao atualizar exame')
      } else {
        router.refresh()
      }
    }

    function handleUploaded(url: string) {
      setLaudoUrl(url)
      router.refresh()
    }

    return (
      <>
        {/* Exames list with checkboxes */}
        {solicitacao.exames && solicitacao.exames.length > 0 && (
          <div className="space-y-3">
            {solicitacao.exames.map((exame) => (
              <div key={exame.id} className="flex items-center gap-3">
                <Checkbox
                  id={exame.id}
                  checked={exame.realizado}
                  onCheckedChange={(checked) =>
                    handleExameToggle(exame.id, checked as boolean)
                  }
                />
                <div>
                  <Label htmlFor={exame.id} className="font-medium cursor-pointer">
                    {exame.nome_exame}
                  </Label>
                  {exame.codigo_procedimento && (
                    <p className="text-xs text-muted-foreground">
                      {exame.codigo_procedimento}
                    </p>
                  )}
                </div>
              </div>
            ))}
          </div>
        )}

        {/* Status buttons */}
        <div className="flex justify-center pt-4">
          <StatusButtons
            solicitacaoId={solicitacao.id}
            currentStatus={solicitacao.status}
            size="default"
            onLiberadoClick={() => setLaudoOpen(true)}
          />
        </div>

        {/* Laudo upload modal */}
        <LaudoUpload
          solicitacaoId={solicitacao.id}
          laudoUrl={laudoUrl}
          open={laudoOpen}
          onOpenChange={setLaudoOpen}
          onUploaded={handleUploaded}
        />
      </>
    )
  }
  ```

- [ ] **Step 2: Create app/solicitacao/[id]/page.tsx**

  ```tsx
  // app/solicitacao/[id]/page.tsx
  import { notFound, redirect } from 'next/navigation'
  import Link from 'next/link'
  import { createClient } from '@/lib/supabase-server'
  import { calcularIdade } from '@/lib/calcularIdade'
  import { ClassificacaoBadge } from '@/components/ClassificacaoBadge'
  import { SolicitacaoClient } from './SolicitacaoClient'
  import { Button } from '@/components/ui/button'
  import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card'
  import { ArrowLeft } from 'lucide-react'

  function formatarCPF(cpf: string | null): string {
    if (!cpf) return '—'
    const digits = cpf.replace(/\D/g, '')
    if (digits.length !== 11) return cpf
    return `${digits.slice(0, 3)}.${digits.slice(3, 6)}.${digits.slice(6, 9)}-${digits.slice(9)}`
  }

  function formatarData(data: string): string {
    const [year, month, day] = data.split('-')
    return `${day}/${month}/${year}`
  }

  export default async function SolicitacaoPage({
    params,
  }: {
    params: { id: string }
  }) {
    const supabase = createClient()

    const { data: { user } } = await supabase.auth.getUser()
    if (!user) redirect('/login')

    const { data: solicitacao } = await supabase
      .from('solicitacoes')
      .select('*, pacientes(*), prestadores(*), exames(*)')
      .eq('id', params.id)
      .single()

    if (!solicitacao) notFound()

    const paciente = solicitacao.pacientes
    const idade = paciente ? calcularIdade(paciente.data_nascimento) : null

    return (
      <div className="min-h-screen bg-gray-50">
        <header className="bg-white border-b px-4 py-3">
          <Button variant="ghost" size="sm" asChild>
            <Link href="/dashboard">
              <ArrowLeft className="h-4 w-4 mr-1" />
              Voltar
            </Link>
          </Button>
        </header>

        <main className="max-w-2xl mx-auto p-4 space-y-4">
          {/* Paciente */}
          <Card>
            <CardHeader>
              <CardTitle className="text-base">Paciente</CardTitle>
            </CardHeader>
            <CardContent className="space-y-1 text-sm">
              <p className="font-semibold text-base">{paciente?.nome_completo ?? '—'}</p>
              {paciente && (
                <p className="text-muted-foreground">
                  {formatarData(paciente.data_nascimento)} · {idade} anos
                </p>
              )}
              <p className="text-muted-foreground">
                CPF: {formatarCPF(paciente?.cpf ?? null)}
              </p>
            </CardContent>
          </Card>

          {/* Solicitação */}
          <Card>
            <CardHeader>
              <CardTitle className="text-base">Solicitação</CardTitle>
            </CardHeader>
            <CardContent className="space-y-1 text-sm">
              <p><span className="text-muted-foreground">Código:</span> {solicitacao.codigo_barras}</p>
              <p><span className="text-muted-foreground">Médico:</span> {solicitacao.medico_solicitante}</p>
              {solicitacao.solicitante_origem && (
                <p><span className="text-muted-foreground">Origem:</span> {solicitacao.solicitante_origem}</p>
              )}
              <p><span className="text-muted-foreground">Prestador:</span> {solicitacao.prestadores?.nome}</p>
              <p>
                <span className="text-muted-foreground">Data prevista:</span>{' '}
                {formatarData(solicitacao.data_prevista)}
                {solicitacao.hora_prevista && ` às ${solicitacao.hora_prevista.slice(0, 5)}`}
              </p>
              <div className="pt-1">
                <ClassificacaoBadge classificacao={solicitacao.classificacao} />
              </div>
            </CardContent>
          </Card>

          {/* Exames + status + upload (client component) */}
          <Card>
            <CardHeader>
              <CardTitle className="text-base">
                Exames ({solicitacao.qtd_exames})
              </CardTitle>
            </CardHeader>
            <CardContent>
              <SolicitacaoClient solicitacao={solicitacao} />
            </CardContent>
          </Card>
        </main>
      </div>
    )
  }
  ```
